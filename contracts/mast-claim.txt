/* MAST Branch 1: House Claims Bet */
/* Player walked away after losing. House claims pessimistic balance after 256 blocks */
/* Security: SIGNEDBY(house key from PREVSTATE) + VERIFYOUT enforces exact payout to addresses in coin state */

ASSERT PREVSTATE(102) EQ 1
ASSERT @COINAGE GTE 256
LET bt=PREVSTATE(108)
IF bt EQ 1 THEN ASSERT SIGNEDBY(PREVSTATE(116))
ELSE ASSERT SIGNEDBY(PREVSTATE(115)) ENDIF
LET a1=PREVSTATE(109) LET a2=PREVSTATE(110)
LET v1=PREVSTATE(111) LET v2=PREVSTATE(112) LET ba=PREVSTATE(103) LET np=PREVSTATE(121)
LET tb=ba*np
IF bt EQ 1 THEN LET p1=v1-tb LET p2=v2+tb
ELSE LET p1=v1+tb LET p2=v2-tb ENDIF
IF p1 GT 0 THEN ASSERT VERIFYOUT(@INPUT a1 p1 @TOKENID FALSE) ENDIF
IF p2 GT 0 THEN ASSERT VERIFYOUT(@INPUT+1 a2 p2 @TOKENID FALSE) ENDIF
RETURN TRUE
